MONGO QUERY FOR DIFF PAIRS

db.getCollection("Net").aggregate([
    {
        $match: {
            $and: [
                { projectId: "66b3e5b7be7179183e4302b7" },
                { name: { $regex: "^(?!STRT)\\w+$", $options: "i" } },
                { name: { $regex: "^(?!.*ENDR$).*", $options: "i" } },
                { name: { $regex: "^(?!.*INCL).+$", $options: "i" } },
                { name: { $regex: "(.*[PNpn].*)", $options: "i" } },
            ],
        },
    },
    {
        $addFields: {
            potentialDiff: {
                $replaceAll: { input: "$name", find: "P", replacement: "N" },
            },
        },
    },
    {
        $project: {
            name: 1,
            potentialDiff: 1,
            matchField: {
                $cond: {
                    if: { $eq: ["$name", "$potentialDiff"] },
                    then: "$name",
                    else: "$potentialDiff",
                },
            },
        },
    },
    {
        $group: {
            _id: "$matchField",
            count: { $sum: 1 },
            items: { $push: "$$ROOT" },
        },
    },
    {
        $match: {
            count: { $eq: 2.0 },
        },
    },
]);
